\begin{definition}
  \emph{(Hermitian Dilation)}
  The Hermitian dilation
  \begin{gather*}
    \mathfrak{H} : \C^{d_1 \times d_2} \to \mathbb{H}_{d_1 \times d_2}
  \end{gather*}
  is a map from a general matrix to an Hermitian matrix defined by
  \begin{gather}
    \label{ rmineq_hermitian_dilation } 
    \mathfrak{H}(B)
    :=
    \begin{bmatrix}
      0   & B \\
      B^* & 0 \\
    \end{bmatrix}
  \end{gather}
\end{definition}
