
%makeindex main.nlo -s nomencl.ist -o main.nls

\nomenclature{$\R$}{set of real numbers}
\nomenclature{$\overline{\R}=\R \cup \left\{ +\infty \right\}$}{extension of the real numbers}
\nomenclature{$\P$}{generic probability measure}
\nomenclature{$\P_X=\P \circ X^{-1}$}{distribution of the random variable $X$}
\nomenclature{$\overset{\mathcal{D}}{\longrightarrow}$}{convergence of distributions}
\nomenclature{$\E[X]$}{expectation of the random variable $X$}
\nomenclature{$\E [X \vert Y]$}{conditional expectation of the random variable $X$ with respect to $\sigma(Y)$}
\nomenclature{$\mathbf{Var}[X]$}{variance of the random variable $X$}
\nomenclature{$x \lor y, x \land y, x^+, x^-$}{maximum, minimum, positive part, negative part of real numbers}
\nomenclature{$\# A$}{cardinality of the set $A$}
\nomenclature{$X\sim \mu$}{the random variable has distribution $\mu$}
