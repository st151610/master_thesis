\begin{theorem}
  \label{bw:cp:th}
  For all
  $i\in \left\{ 1,\ldots,n \right\}$
  and all
  $\varepsilon>0$ it holds
  \begin{gather}
    \P
    \left[ 
   \left| 
   1 / \pi_i
   -
   w_i
   \right|
    \ge
    \varepsilon
    \right]
    \ 
    \to
    \ 
    0
    \qquad
    \text{for}\ 
    N
    \to 
    \infty
    \,.
  \end{gather}
\end{theorem}

\begin{proof}
  It holds
 \begin{align*}
   \left| 
   1 / \pi_i
   -
   w_i
   \right|
   & 
   \ 
   =
   \ 
   \left| 
   1 / \pi_i
   -
   (
      f ^ { ' }
   )
   ^ {-1}
   \left( 
   \inner
   { B(X_i)}
   { \lambda ^ \dagger }
   +
   \lambda^\dagger_0
   \right)
   \right|
   %%%% 1 %%%%
   \\
   & 
   \ 
   \le
   \ 
   \left| 
   1 / \pi_i
   -
   (
      f ^ { ' }
   )
   ^ {-1}
   \left( 
   \inner
   { B(X_i)}
   { \lambda ^ * }
   +
   \lambda^\dagger_0
   \right)
   \right|
      %%%% 1 a %%%%
   \\
   &
   \qquad
   +
   \ 
\left| 
   (
      f ^ { ' }
   )
   ^ {-1}
   \left( 
   \inner
   { B(X_i)}
   { \lambda ^ \dagger }
   +
   \lambda^\dagger_0
   \right)
   -
   (
      f ^ { ' }
   )
   ^ {-1}
   \left( 
   \inner
   { B(X_i)}
   { \lambda ^ *}
   +
   \lambda^\dagger_0
   \right)
   \right|
      %%%% 2 %%%%
%   \\
%   &
%   \ 
%   =:
%   I_1 +
%   I_2
 \end{align*}
 By the universal consistency of $B$ and the continuity of $(f^{'})^{-1}$ the first term converges to $0$ in probability.
 By the continuity of $(f^{'})^{-1}$,
 the uniform boundedness of $\norm{B}$ and the dual consistency
 the second term converges to $0$ in probability.

\end{proof}
