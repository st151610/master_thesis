\begin{assumption}
  The set of (measurable) basis functions 
\begin{gather*}
  \mathfrak{B}
  :
  =
  \left\{ 
B_k\colon (\mathcal{X},\Sigma_\mathcal{X})\to(\R,\mathcal{B}(\R))
\ 
|
\ 
k\in \left\{ 1,\ldots,\# \mathfrak{B} \right\}
  \right\}
\end{gather*}
  satisfies
  \begin{itemize}
    \item
      $
      \sum_{k=1}^{\# B} 
      B_k(X_i)
      \ 
      =
      \ 
      1
      $
      for all 
      $i\in \left\{ 1,\ldots ,N \right\}$
    \item
      $
      \sum_{k=1}^{\# B} 
      B_k^2(x)
      \ 
      \lesssim
      \ 
      1
      $
      for all $x\in\mathcal{X}$.
    \item
      $
      \norm{X_k - X_i}_2
      \cdot
      B_k(X_i)
      \to
      0
      \ 
      $
      for all 
      $
      (i,k)
      \in
      \left\{ 1,\ldots, N \right\}
      \times
      \left\{ 1,\ldots,\# B \right\}
      $
      and
      $N\to\infty$
  \end{itemize}
\end{assumption}
